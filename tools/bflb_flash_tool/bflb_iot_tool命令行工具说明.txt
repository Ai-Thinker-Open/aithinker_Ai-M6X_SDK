--chipname：芯片型号，必填
--interface：默认uart
--port：com口，默认自动查找
--baudrate：波特率默认115200
--xtal：xtal
--dts：device tree文件
--pt：partition table文件
--firmware：firmware bin文件，必填 
--boot2：boot2 bin文件 
--mfg：mfg bin文件 
--romfs：romfs文件夹 
--build：只生成镜像
--erase：擦除flash
--single：单文件烧写
--addr：烧写地址默认0
--config：eflash_loader_cfg.ini
--ota：指定ota文件生成目录
